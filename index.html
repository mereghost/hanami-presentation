Validações

São providas por um modulo que pode ser incluido em qualquer objeto.

- Falando primeiro porque isso aparece em vários pontos. - Mecanismo de includes vai ser o padrão em muitos lugares

Os validadores usam o dry-validation por baixo

Um exemplo simples


# Rails
validates :some_field, presence: true, length: { in: 3..8 }

# Hanami
required(:some_field) { filled? & str? & size?(3..8)  }
# Macros!
required(:some_field).filled(:str?, size?: (3..8))

Custom Predicates


class AnExampleClass
  include Hanami::Validations
  EMAIL_REGEX = /.+@.+/ #totally works!

  predicate :email?, message: 'not an email' do |actual|
    actual =~ EMAIL_REGEX
  end

  validations do
    required(:nullable).maybe(:int?)
    required(:something_or_the_other) { str? | int? }
    required(:an_email) { email? }
    optional(:an_array).each(:int?)
  end
end
Apresentação mais concisa, poderosa e clara das validações. Campos opcionais e tal.

Mas eu chamo um método sempre a cada request!

module Web
  module MyReusableBehaviour
  def self.included(action)
    action.class_eval do
      before :behaviour
      expose :some_object
    end
  end
  private def behaviour; @some_object = "Nothing to see here"; end
end

module Web::Controllers::AwesomeClassName
  class Index
    include Web::Action
    include Web::MyReusableBehaviour
  end
end

Repositories

São a interface entre o BD e a Entity.


class AwesomeThingRepository
  include Hanami::Repository
end

Mas o que dá pra fazer com eles?

  • #all, #find, #create, #update, #delete, #first, #last
  • #persist
  • #clear
  • #execute, #fetch

O método #query é privado

Queries com valor de negócio

AwesomeThing.order(:created_at).limit(10)

class AwesomeThingRepository
  include Hanami::Repository
  def self.most_recent(amount = 10)
    query do
      order(:created_at).limit(amount)
    end
  end
end

Mappings

É como ligamos os repositorios, ao DB e às entities


Hanami::Model.configure do
  adapter type: :sql, uri: ENV['DATABASE_URL']
  migrations 'db/migrations'
  schema 'db/schema.sql'
  mapping do
    collection :awesome_things do
      entity AwesomeThing
      # repository AwesomeThingRepository
      attribute :id, Integer
      attribute :coolness_index, Integer
      attribute :created_at, DateTime
      attribute :updated_at, DateTime
    end
  end
end
